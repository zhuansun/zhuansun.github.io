explain
SELECT t1.process_status AS STATUS,
       count(t1.id)         cnt
FROM amp_event_log t1
JOIN
     (select * from amp_site where sf_site = 0) 
    t3 ON t3.NAME = t1.exception_subject
JOIN ( SELECT event_code,deal_organization_id FROM amp_event WHERE is_history = 0 AND deal_organization_id = 1024) 
    t2 ON t1.event_code = t2.event_code
where 1 = 1
  and t1.happen_time >= '2022-3-2 20:37:04'
  and t1.happen_time <= '2023-3-2 20:37:13'
  and t1.prov in ('浙江省', '广东省')
  and t1.city in ('杭州市', '深圳市')
GROUP BY process_status;

分析语句

Using where; Using temporary; Using filesort

Using where

Range checked for each record (index map: 0x2)

  • Using temporary

    • group by 语句一般会使用内部临时表，因为需要使用内部临时表存储数量；
    • 一般情况下group by都是会使用临时表的，但是再有些业务场景下，也是可以优化成不用临时表的，所以效率会更高
    • 【优化点一】：是否可以将这个 group by 语句取消使用内部临时表

  • Using filesort

    • group by 分组后默认自然顺序排序，如果不需要排序，可以改成：order by null；（8.0版本group by已经不会排序了）
    • 所以可以根据具体的业务场景看，是否需要排序
    • 【优化点二】：group by 语句是否可以不需要排序

  • Using join buffer (Block Nested Loop)

    • 使用了join语句，而且是BNL算法，效率很低，首先看这个语句是否可以不用join

    • 【优化点三】：是否可以避免join？从业务代码的角度进行优化？

    • 如果join优化不了，是否可以将BNL算法优化成NLJ（BKA算法）

    • 【优化点四】：是否可以将BNL算法优化成NLJ（BKA算法）

    • 但是呢，即使使用了NLJ（BKA算法），效率还是不如hash-join（mysql8.0支持hash-join，效率高）

    • 【优化点五】：是否可以通过业务代码实现hash-join

select log.process_status as STATUS,
       count(log.id)         cnt
from amp_event event left join amp_event_log log on event.event_code=log.event_code

  and event.deal_organization_id=1024
  and log.prov in ('浙江省', '广东省')
  and log.city in ('杭州市', '深圳市')
GROUP BY log.process_status order by null;

explain
SELECT t1.event_code,
       t2.event_name     AS name,
       t1.process_status AS STATUS,
       t1.cnt
FROM (
         SELECT t.event_code,
                t.process_status,
                t.exception_subject,
                count(t.id) cnt
         FROM amp_event_log t
                  JOIN
              (select *
               from amp_site
              ) t3
              ON t3.NAME = t.exception_subject
         where 1 = 1
           and t.happen_time >= '2022-3-2 20:37:04'
           and t.happen_time <= '2023-3-2 20:37:13'
           and t.prov in ('浙江省', '广东省')
           and t.city in ('杭州市', '深圳市')
         GROUP BY t.event_code, t.process_status
     ) t1
         JOIN (
    SELECT event_code,
           event_name,
           deal_organization_id
    FROM amp_event
    WHERE is_history = 0
      AND deal_organization_id = 1024
) t2 ON t1.event_code = t2.event_code;

select t2.event_code as event_code, t1.event_name as name, t2.process_status as status, COUNT(t2.id) as cnt
from amp_event t1
         left join amp_event_log t2 on t1.event_code = t2.event_code
where 1 = 1
  and t2.happen_time >= '2022-3-2 20:37:04'
  and t2.happen_time <= '2023-3-2 20:37:13'
  and t2.prov in ('浙江省', '广东省')
  and t2.city in ('杭州市', '深圳市')
  and t1.is_history = 0
  and t1.deal_organization_id = 1024
group by t2.event_code, t2.process_status
